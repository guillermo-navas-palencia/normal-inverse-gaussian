\begin{align*}
A &= (k + 1) k z^2 t_+^2\\
B &= (k + 1) z \left(z \beta^2 + (\lambda + 1 + 2k) t_+\right)\\
C &= (k + 1) (\lambda + 1 + k) - z^2 (t_+^2 - \beta^2) + z t_+ (\lambda + 1)\\
D &= \lambda + 1 - 2zt_+
\end{align*}
